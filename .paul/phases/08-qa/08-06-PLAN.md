**PluginEditor.cpp** — add `timerCallback()` implementation:
```cpp
void GranoAudioProcessorEditor::timerCallback()
{
    // Grain count from engine
    std::array<GranularEngine::GrainSnapshot, GranularEngine::MaxActiveGrains> snaps;
    const int grains = processorRef.getEngine().getGrainSnapshots(
        snaps.data(), GranularEngine::MaxActiveGrains);
    footerGrainLabel_.setText(juce::String(grains) + " grains",
                               juce::dontSendNotification);

    // CPU load from JUCE host context
    const double cpu = processorRef.getCpuUsage();
    footerCpuLabel_.setText("CPU " + juce::String(cpu * 100.0, 1) + "%",
                              juce::dontSendNotification);

    // MASTER dB readout
    const float vol = processorRef.getAPVTS()
                          .getRawParameterValue(ParamIDs::masterVolume)->load();
    const float dB = juce::Decibels::gainToDecibels(vol, -60.0f);
    masterDbLabel_.setText(juce::String(dB, 1) + " dB",
                            juce::dontSendNotification);
}
```
