ModuleTabPanel::ModuleTabPanel()
{
    for (int i = 0; i < kTabCount; ++i)
    {
        tabBtns_[i].setButtonText(kTabLabels[i]);
        tabBtns_[i].setClickingTogglesState(false);
        tabBtns_[i].onClick = [this, i] { showTab(i); };
        addAndMakeVisible(tabBtns_[i]);
    }
    // Populate comboboxes
    subGrainDepthBox_.addItem("Depth 0",  1);
    subGrainDepthBox_.addItem("Depth 1",  2);
    subGrainDepthBox_.addItem("Depth 2",  3);

    stochasticDistBox_.addItem("Uniform",      1);
    stochasticDistBox_.addItem("Gaussian",     2);
    stochasticDistBox_.addItem("Poisson",      3);
    stochasticDistBox_.addItem("Exponential",  4);
    stochasticDistBox_.addItem("Pareto",       5);
    stochasticDistBox_.addItem("1/f",          6);

    spectralModeBox_.addItem("Freeze", 1);
    spectralModeBox_.addItem("Blur",   2);

    populateTriggerBox(triggerModeBox_);
    populateQuantizeBox(quantizeScaleBox_);
}

void ModuleTabPanel::init(juce::AudioProcessorValueTreeState& apvts)
{
    buildEngineTab  (apvts);
    buildMotionTab  (apvts);
    buildColorTab   (apvts);
    buildPatternTab (apvts);

    lfoPanel_.init(apvts);
    addAndMakeVisible(lfoPanel_);
    lfoPanel_.setVisible(false);

    modMatrixView_.init(apvts);
    addAndMakeVisible(modMatrixView_);
    modMatrixView_.setVisible(false);

    showTab(0);
}

void ModuleTabPanel::paint(juce::Graphics& g)
{
    const auto b = getLocalBounds();

    // Background
    g.setColour(kSurface);
    g.fillRect(b);

    // Top border
    g.setColour(kBorder);
    g.drawLine(0.f, 0.f, (float)b.getWidth(), 0.f, 1.f);

    // Active tab accent — 2px mint line below the active tab button
    const int tabW = b.getWidth() / kTabCount;
    g.setColour(kMint);
    g.fillRect(activeTab_ * tabW, kTabBarH - 2, tabW, 2);
}

void ModuleTabPanel::resized()
{
    const auto b     = getLocalBounds();
    const int  w     = b.getWidth();
    const int  tabW  = w / kTabCount;
    const int  contentY = kTabBarH;
    const int  contentH = b.getHeight() - kTabBarH;
    const int  pad   = 8;

    // Tab buttons
    for (int i = 0; i < kTabCount; ++i)
        tabBtns_[i].setBounds(i * tabW, 0, tabW, kTabBarH);

    const juce::Rectangle<int> content { pad, contentY + pad,
                                          w - 2 * pad, contentH - 2 * pad };

    if (activeTab_ == 0)  // ENGINE
    {
        // Left column: Sub-grain + Stochastic comboboxes
        // Right column: Feedback section + Spectral section
        const int halfW = content.getWidth() / 2;
        const int comboH = 22;
        const int labelH = 16;
        const int knobH  = 70;

        // Left — sub-grain depth
        subGrainLabel_.setBounds  (content.getX(), content.getY(), halfW, labelH);
        subGrainDepthBox_.setBounds(content.getX(), content.getY() + labelH, halfW - pad, comboH);

        // Left — stochastic distribution
        const int stocY = content.getY() + labelH + comboH + pad;
        stochasticLabel_.setBounds  (content.getX(), stocY, halfW, labelH);
        stochasticDistBox_.setBounds(content.getX(), stocY + labelH, halfW - pad, comboH);

        // Right — Feedback toggle + 2 knobs
        const int rx = content.getX() + halfW;
        feedbackEnableBtn_.setBounds(rx, content.getY(), halfW / 2, comboH);
        feedbackGainKnob_.setBounds (rx,           content.getY() + comboH + pad, halfW / 2, knobH);
        feedbackDampKnob_.setBounds (rx + halfW/2, content.getY() + comboH + pad, halfW / 2, knobH);

        // Right — Spectral toggle + mode + blur knob
        const int spY = content.getY() + comboH + pad + knobH + pad;
        spectralEnableBtn_.setBounds(rx, spY, halfW / 3, comboH);
        spectralModeLabel_.setBounds(rx + halfW / 3, spY, halfW / 4, labelH);
        spectralModeBox_.setBounds  (rx + halfW / 3, spY + labelH, halfW / 3 - pad, comboH - labelH);
        spectralBlurKnob_.setBounds (rx + halfW * 2/3, spY, halfW / 3, knobH);
    }
    else if (activeTab_ == 1)  // MOTION
    {
        // Enable toggle top-left
        motionEnableBtn_.setBounds(content.getX(), content.getY(), 100, 22);

        // Two rows of 4 knobs each
        const int knobH = (content.getHeight() - 22 - pad) / 2;
        const int knobW = content.getWidth() / 4;
        const int row1Y = content.getY() + 22 + pad;
        const int row2Y = row1Y + knobH;
        const int cx    = content.getX();

        wowDepthKnob_.setBounds    (cx + knobW * 0, row1Y, knobW, knobH);
        wowRateKnob_.setBounds     (cx + knobW * 1, row1Y, knobW, knobH);
        flutterDepthKnob_.setBounds(cx + knobW * 2, row1Y, knobW, knobH);
        flutterRateKnob_.setBounds (cx + knobW * 3, row1Y, knobW, knobH);

        driftAmountKnob_.setBounds (cx + knobW * 0, row2Y, knobW, knobH);
        crackleLevelKnob_.setBounds(cx + knobW * 1, row2Y, knobW, knobH);
        crackleColorKnob_.setBounds(cx + knobW * 2, row2Y, knobW, knobH);
        // knobW * 3 cell intentionally empty in row 2
    }
    else if (activeTab_ == 2)  // COLOR
    {
        colorEnableBtn_.setBounds(content.getX(), content.getY(), 100, 22);

        const int knobH = content.getHeight() - 22 - pad;
        const int knobW = content.getWidth() / 4;
        const int ky    = content.getY() + 22 + pad;
        const int cx    = content.getX();

        saturateKnob_.setBounds(cx + knobW * 0, ky, knobW, knobH);
        decimateKnob_.setBounds(cx + knobW * 1, ky, knobW, knobH);
        tiltEqKnob_.setBounds  (cx + knobW * 2, ky, knobW, knobH);
        verbMixKnob_.setBounds (cx + knobW * 3, ky, knobW, knobH);
    }
    else if (activeTab_ == 3)  // PATTERN
    {
        patternEnableBtn_.setBounds(content.getX(), content.getY(), 110, 22);

        euclidPulsesKnob_.setBounds (cx + knobW * 3, knobY, knobW, knobH);
        euclidStepsKnob_.setBounds  (cx + knobW * 4, knobY, knobW, knobH);
        euclidRotKnob_.setBounds    (cx + knobW * 5, knobY, knobW, knobH);
    }
    else if (activeTab_ == 4)  // LFO
    {
        lfoPanel_.setBounds(content);

    else if (activeTab_ == 5)  // MOD

        modMatrixView_.setBounds(content);
