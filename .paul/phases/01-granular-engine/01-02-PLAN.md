  <action>
    **GranularEngine.h** — declare the class. No juce includes in the header beyond forwarding headers needed by the type signatures.

    ```cpp
    #pragma once
    #include "GrainPool.h"
    #include <juce_audio_basics/juce_audio_basics.h>
    #include <juce_core/juce_core.h>
    #include <array>
    #include <atomic>
    #include <vector>

    class GranularEngine
    {
    public:
        static constexpr int FifoCapacity    = 512;
        static constexpr int MaxActiveGrains = 256;

        GranularEngine();
        ~GranularEngine();

        void prepare(double sampleRate, int samplesPerBlock) noexcept;
        void processBlock(juce::AudioBuffer<float>& buffer) noexcept;
        void reset() noexcept;

    private:
        // --- inner scheduler thread ---
        class SchedulerThread : public juce::Thread
        {
        public:
            SchedulerThread(GranularEngine& e) : juce::Thread("GrainScheduler"), engine_(e) {}
            void run() override;
        private:
            GranularEngine& engine_;
        };

        void scheduleGrain() noexcept;  // called by SchedulerThread only
        bool renderGrain(Grain* g, float* L, float* R, int numSamples) noexcept; // returns true when grain finished

        // --- data members ---
        GrainPool pool_;

        // SPSC FIFO: SchedulerThread → audio thread
        juce::AbstractFifo             fifo_{ FifoCapacity };
        std::array<Grain*, FifoCapacity> fifoSlots_{};

        // Active grains on audio thread (no sharing with other threads)
        std::array<Grain*, MaxActiveGrains> activeGrains_{};
        int                                 activeGrainCount_{ 0 };

        // Hard-coded test sample (sine wave, allocated once in prepare())
        std::vector<float> testSample_;
        std::atomic<bool>  sampleReady_{ false };

        double sampleRate_{ 48000.0 };

        SchedulerThread schedulerThread_;
    };
    ```

    **GranularEngine.cpp** — implement:

    1. **Constructor/Destructor:**
       - Constructor: no work (SchedulerThread not started yet).
       - Destructor: call `schedulerThread_.stopThread(2000)`.

    2. **prepare():**
       - Store sampleRate.
       - Generate hard-coded 440 Hz sine wave, 1 second duration:
         ```cpp
         const int len = (int)sampleRate;
         testSample_.resize(len + 2); // +2 guard samples for interpolation boundary
         for (int i = 0; i < len + 2; ++i)
             testSample_[i] = std::sin(2.0f * juce::MathConstants<float>::pi * 440.0f * (float)i / (float)sampleRate);
         sampleReady_.store(true, std::memory_order_release);
         ```
       - Start scheduler thread: `schedulerThread_.startThread()`.

    3. **reset():**
       - Call `schedulerThread_.stopThread(500)`.
       - Release all active grains back to pool.
       - Reset `activeGrainCount_ = 0`.
       - Reset `fifo_` by draining it.

    4. **SchedulerThread::run():**
       ```cpp
       while (!threadShouldExit())
       {
           engine_.scheduleGrain();
           wait(20); // 50 grains/sec ≈ 1 grain every 20 ms
       }
       ```

    5. **scheduleGrain():**
       - Return early if `!sampleReady_.load(std::memory_order_acquire)`.
       - `Grain* g = pool_.acquire();` — return if nullptr (pool exhausted).
       - Fill grain:
         ```cpp
         const int grainLen = (int)(0.1 * sampleRate_); // 100 ms
         g->source        = testSample_.data();
         g->startPos      = 0;
         g->lengthSamples = grainLen;
         g->pitchRatio    = 1.0f;
         g->pan           = 0.0f;
         g->shape         = EnvelopeShape::Hann;
         g->currentPhase  = 0.0f;
         ```
       - Push to FIFO using AbstractFifo:
         ```cpp
         int s1, n1, s2, n2;
         fifo_.prepareToWrite(1, s1, n1, s2, n2);
         if (n1 > 0) { fifoSlots_[s1] = g; fifo_.finishedWrite(1); }
         else if (n2 > 0) { fifoSlots_[s2] = g; fifo_.finishedWrite(1); }
         else { pool_.release(g); } // FIFO full — drop grain
         ```

    6. **processBlock():**
       - If `buffer.getNumChannels() < 1`: return.
       - Drain FIFO into activeGrains_:
         ```cpp
         int s1, n1, s2, n2;
         fifo_.prepareToRead(fifo_.getNumReady(), s1, n1, s2, n2);
         auto addGrain = [&](Grain* g) {
             if (activeGrainCount_ < MaxActiveGrains) activeGrains_[activeGrainCount_++] = g;
             else pool_.release(g); // no room — drop
         };
         for (int i = 0; i < n1; ++i) addGrain(fifoSlots_[s1 + i]);
         for (int i = 0; i < n2; ++i) addGrain(fifoSlots_[s2 + i]);
         fifo_.finishedRead(n1 + n2);
         ```
       - Get write pointers (handle mono/stereo):
         ```cpp
         float* L = buffer.getWritePointer(0);
         float* R = buffer.getWritePointer(buffer.getNumChannels() > 1 ? 1 : 0);
         const int N = buffer.getNumSamples();
         ```
       - Render and cull active grains (swap-erase pattern, no allocation):
         ```cpp
         for (int gi = 0; gi < activeGrainCount_; )
         {
             bool done = renderGrain(activeGrains_[gi], L, R, N);
             if (done) { pool_.release(activeGrains_[gi]); activeGrains_[gi] = activeGrains_[--activeGrainCount_]; }
             else { ++gi; }
         }
         ```

    7. **renderGrain():**
       - For each output sample index `n` in [0, numSamples):
         ```cpp
         const float phase = g->currentPhase;
         const float srcPosF = g->startPos + phase * g->lengthSamples * g->pitchRatio;
         const int   srcInt  = (int)srcPosF;
         const float frac    = srcPosF - (float)srcInt;
         // Bounds check: if srcInt+1 >= source length, clamp
         const int srcLen = g->lengthSamples; // use lengthSamples as proxy for source bounds
         // Actually use testSample_.size() — but renderGrain doesn't own that.
         // Instead: clamp srcInt to [0, (int)g->lengthSamples * 4] — safe with guard samples
         const float s0 = g->source[srcInt];
         const float s1 = g->source[srcInt + 1]; // guard samples prevent OOB
         const float s  = s0 + frac * (s1 - s0);
         const float env = applyEnvelope(g->shape, phase);
         const float L_gain = 0.5f * (1.0f - g->pan);
         const float R_gain = 0.5f * (1.0f + g->pan);
         L[n] += s * env * L_gain;
         R[n] += s * env * R_gain;
         g->currentPhase += 1.0f / (float)g->lengthSamples;
         if (g->currentPhase >= 1.0f) return true; // finished mid-block is fine
         ```
       - Return `false` if loop completes without phase reaching 1.

    **RT-safety invariants to enforce:**
    - `processBlock` and `renderGrain`: no `new`, no `delete`, no `std::vector` growth, no mutex, no sleep, no I/O.
    - Source pointer bounds: `testSample_` has +2 guard samples so `srcInt + 1` never goes OOB for any phase in [0, 1] with pitchRatio ≤ 2.0. For 01-02 pitchRatio is always 1.0f so this is safe. Document this constraint.
    - `scheduleGrain()` runs only on SchedulerThread, never on audio thread.

    Include `<juce_core/juce_core.h>` and `<juce_audio_basics/juce_audio_basics.h>` in the .cpp. Do NOT include all of JUCE — module-level includes only.

    ```cpp
    // ── Pitch ratio math ──────────────────────────────────────────────────────────

    // Helper: given a grain at startPos=0, lengthSamples=N, pitchRatio=r,
    // compute the source read position at a given phase.
    static float srcPosAtPhase(float phase, int grainLen, float pitchRatio)
    {
        return phase * (float)grainLen * pitchRatio;
    }

    TEST_CASE("Pitch ratio 1.0 reads exactly lengthSamples source frames", "[grain][pitch]")
    {
        const int N = 200;
        REQUIRE(srcPosAtPhase(0.0f, N, 1.0f) == Catch::Approx(0.0f));
        REQUIRE(srcPosAtPhase(1.0f, N, 1.0f) == Catch::Approx((float)N));
        REQUIRE(srcPosAtPhase(0.5f, N, 1.0f) == Catch::Approx(100.0f));
    }

    TEST_CASE("Pitch ratio 2.0 reads twice as many source frames (pitch up)", "[grain][pitch]")
    {
        const int N = 200;
        REQUIRE(srcPosAtPhase(1.0f, N, 2.0f) == Catch::Approx(400.0f));
        REQUIRE(srcPosAtPhase(0.5f, N, 2.0f) == Catch::Approx(200.0f));
    }

    TEST_CASE("Pitch ratio 0.5 reads half as many source frames (pitch down)", "[grain][pitch]")
    {
        const int N = 200;
        REQUIRE(srcPosAtPhase(1.0f, N, 0.5f) == Catch::Approx(100.0f));
        REQUIRE(srcPosAtPhase(0.5f, N, 0.5f) == Catch::Approx(50.0f));
    }

    TEST_CASE("Pitch ratio 1.5 (fractional) source position is linear", "[grain][pitch]")
    {
        const int N = 100;
        REQUIRE(srcPosAtPhase(0.0f, N, 1.5f) == Catch::Approx(0.0f));
        REQUIRE(srcPosAtPhase(1.0f, N, 1.5f) == Catch::Approx(150.0f));
        REQUIRE(srcPosAtPhase(0.25f, N, 1.5f) == Catch::Approx(37.5f));
    }
    ```

    These tests verify the source-position formula used in `renderGrain` without needing live audio. No GranularEngine instantiation required in the test — the formula is pure math.
