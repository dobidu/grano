## T1 — MultiSampleBank.{h,cpp} + 4 params + CMake
type: auto

### Context
MultiSampleBank wraps 4 SampleBuffer slots. Each slot uses the same hot-swap pattern as
the existing SampleBuffer (pending → current → retired). The key addition is `pickSlot()`:
a weighted-random selector that runs on the scheduler thread — must be RT-safe (no alloc,
no locks; uses only stack arrays and SampleBuffer's atomic getReadPointer/getNumSamples).

`trySwapAll()` returns true if ANY slot swapped — the caller (GranularEngine::processBlock)
uses this to flush active grains whose source pointers are now stale.

`getSlot(int)` returns a SampleBuffer reference so PluginProcessor can expose slot 0 to
WaveformDisplay and spectral trigger without changing the existing interface.

**MultiSampleBank.h:**
```cpp
#pragma once
#include "SampleBuffer.h"
#include <juce_core/juce_core.h>
#include <array>
#include <utility>

// MultiSampleBank — 4-slot SampleBuffer wrapper with per-grain weighted selection.
//
// Load contract (same as SampleBuffer):
//   loadSlot()       — message/loading thread
//   trySwapAll()     — audio thread, once per processBlock
//   processRetiredAll() — message thread (~30 Hz Timer)
//   pickSlot()       — scheduler thread (RT-safe)
class MultiSampleBank
{
public:
    static constexpr int kNumSlots = 4;

    MultiSampleBank() = default;

    // Message/loading thread: load a buffer into a slot.
    void loadSlot(int slot,
                  std::unique_ptr<juce::AudioBuffer<float>> buf,
                  int numSamples);

    // Audio thread: try-swap all pending buffers.
    // Returns true if any slot swapped (caller should flush active grains).
    [[nodiscard]] bool trySwapAll() noexcept;

    // Message thread (~30 Hz): delete retired buffers for all slots.
    void processRetiredAll();

    // Scheduler thread: weighted-random slot selection. RT-safe.
    // weights[kNumSlots] — unnormalized; slots with weight=0 or no data excluded.
    // Returns {ptr, len} for the selected slot, or {nullptr, 0} if none available.
    std::pair<const float*, int> pickSlot(juce::Random& rng,
                                          const float weights[kNumSlots]) const noexcept;

    // Direct per-slot read access (any thread, same safety as SampleBuffer).
    const float* getReadPointer(int slot) const noexcept;
    int          getNumSamples(int slot)  const noexcept;

    // Reference to a specific slot's SampleBuffer (e.g. slot 0 for WaveformDisplay).
    SampleBuffer& getSlot(int slot) noexcept { return slots_[slot]; }

private:
    std::array<SampleBuffer, kNumSlots> slots_;

    JUCE_DECLARE_NON_COPYABLE_WITH_LEAK_DETECTOR(MultiSampleBank)
};
```

**MultiSampleBank.cpp:**
```cpp
#include "MultiSampleBank.h"

void MultiSampleBank::loadSlot(int slot,
                                std::unique_ptr<juce::AudioBuffer<float>> buf,
                                int numSamples)
{
    if (slot < 0 || slot >= kNumSlots)
        return;
    slots_[slot].setPending(std::move(buf), numSamples);
}

bool MultiSampleBank::trySwapAll() noexcept
{
    bool anySwapped = false;
    for (auto& slot : slots_)
        if (slot.trySwap())
            anySwapped = true;
    return anySwapped;
}

void MultiSampleBank::processRetiredAll()
{
    for (auto& slot : slots_)
        slot.processRetired();
}

std::pair<const float*, int> MultiSampleBank::pickSlot(
    juce::Random& rng,
    const float weights[kNumSlots]) const noexcept
{
    // Build cumulative weight table for valid (loaded + weight>0) slots.
    float cumulative[kNumSlots];
    int   validIdx[kNumSlots];
    float total = 0.f;
    int   count = 0;

    for (int s = 0; s < kNumSlots; ++s)
    {
        const float w   = weights[s];
        const int   len = slots_[s].getNumSamples();
        if (w > 0.f && len > 0)
        {
            total += w;
            cumulative[count] = total;
            validIdx[count]   = s;
            ++count;
        }
    }

    if (count == 0 || total <= 0.f)
        return { nullptr, 0 };

    const float pick = rng.nextFloat() * total;
    for (int i = 0; i < count; ++i)
    {
        if (pick <= cumulative[i] || i == count - 1)
        {
            const int s = validIdx[i];
            return { slots_[s].getReadPointer(), slots_[s].getNumSamples() };
        }
    }
    return { nullptr, 0 };
}

const float* MultiSampleBank::getReadPointer(int slot) const noexcept
{
    if (slot < 0 || slot >= kNumSlots) return nullptr;
    return slots_[slot].getReadPointer();
}

int MultiSampleBank::getNumSamples(int slot) const noexcept
{
    if (slot < 0 || slot >= kNumSlots) return 0;
    return slots_[slot].getNumSamples();
}
```

#include <catch2/catch_test_macros.hpp>
#include <memory>
#include <cmath>

#include "Engine/MultiSampleBank.h"

static std::unique_ptr<juce::AudioBuffer<float>> makeBuf(int frames, float value = 0.5f)
{
    auto buf = std::make_unique<juce::AudioBuffer<float>>(1, frames + 2);
    buf->clear();
    for (int i = 0; i < frames; ++i)
        buf->getWritePointer(0)[i] = value;
    return buf;
}

static void loadAndSwap(MultiSampleBank& bank, int slot, int frames, float value = 0.5f)
{
    bank.loadSlot(slot, makeBuf(frames, value), frames);
    bank.trySwapAll();
}

TEST_CASE("MultiSampleBank empty bank returns null from pickSlot", "[multisample]")
{
    MultiSampleBank bank;
    juce::Random rng;
    float weights[4] = { 1.f, 0.f, 0.f, 0.f };
    auto [ptr, len] = bank.pickSlot(rng, weights);
    REQUIRE(ptr == nullptr);
    REQUIRE(len == 0);

TEST_CASE("MultiSampleBank loaded slot 0 is accessible", "[multisample]")
{
    MultiSampleBank bank;
    loadAndSwap(bank, 0, 512);
    REQUIRE(bank.getReadPointer(0) != nullptr);
    REQUIRE(bank.getNumSamples(0) == 512);
}

TEST_CASE("MultiSampleBank pickSlot returns data from loaded slot", "[multisample]")
{
    MultiSampleBank bank;
    loadAndSwap(bank, 0, 256);
    juce::Random rng;
    float weights[4] = { 1.f, 0.f, 0.f, 0.f };
    auto [ptr, len] = bank.pickSlot(rng, weights);
    REQUIRE(ptr != nullptr);
    REQUIRE(len == 256);

    loadAndSwap(bank, 1, 256);

TEST_CASE("MultiSampleBank weighted selection proportions", "[multisample]")
{
    MultiSampleBank bank;
    loadAndSwap(bank, 0, 128, 0.1f);
    loadAndSwap(bank, 1, 128, 0.9f);
    juce::Random rng(123);
    float weights[4] = { 3.f, 1.f, 0.f, 0.f };

    const float* ptr0 = bank.getReadPointer(0);
    int hits0 = 0;
    const int N = 10000;
    for (int i = 0; i < N; ++i)
    {
        auto [ptr, len] = bank.pickSlot(rng, weights);
        if (ptr == ptr0) ++hits0;
    }

    // Expected ~75% for slot 0 (weight 3 out of 4 total); allow ±6%
    const float ratio = static_cast<float>(hits0) / N;
    REQUIRE(ratio > 0.69f);
    REQUIRE(ratio < 0.81f);
}
