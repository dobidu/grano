**Source/UI/CurveEditor.cpp:**
```cpp
#include "CurveEditor.h"

const char* CurveEditor::shapeLabels_[kNumShapes] = {
    "HANN", "TUKEY", "GAUSS", "TRI", "RECT"
};

CurveEditor::CurveEditor()
{
    startTimerHz(30);
}

CurveEditor::~CurveEditor()
{
    stopTimer();
}

void CurveEditor::resized()
{
    const int w = getWidth() / kNumShapes;
    for (int i = 0; i < kNumShapes; ++i)
        thumbnailBounds_[i] = { i * w, 0, w, getHeight() };
}

void CurveEditor::paint(juce::Graphics& g)
{
    // Read current shape
    int currentShape = 0;
    if (apvts_ != nullptr)
    {
        if (auto* p = apvts_->getRawParameterValue(ParamIDs::envelopeShape))
            currentShape = (int)std::clamp(p->load(std::memory_order_relaxed), 0.0f, 4.0f);
    }

    for (int i = 0; i < kNumShapes; ++i)
    {
        const bool selected = (i == currentShape);
        const auto b = thumbnailBounds_[i].toFloat().reduced(3.0f);

        // Background
        g.setColour(selected ? juce::Colour(0xff0e1a15) : juce::Colour(0xff1c1f26));
        g.fillRoundedRectangle(b, 3.0f);

        // Border
        g.setColour(selected ? juce::Colour(0xff7CF5C4) : juce::Colour(0xff3A3D45));
        g.drawRoundedRectangle(b, 3.0f, 1.0f);

        // Envelope curve
        const auto curveArea = b.reduced(4.0f, 8.0f);
        drawEnvelopeCurve(g, static_cast<EnvelopeShape>(i), curveArea, selected);

        // Label
        g.setColour(selected ? juce::Colour(0xff7CF5C4) : juce::Colour(0xff888ea8));
        g.setFont(10.0f);
        g.drawText(shapeLabels_[i],
                   thumbnailBounds_[i].toFloat().withTrimmedTop(thumbnailBounds_[i].getHeight() - 14.0f),
                   juce::Justification::centred, false);
    }
}

void CurveEditor::drawEnvelopeCurve(juce::Graphics& g, EnvelopeShape shape,
                                     juce::Rectangle<float> bounds, bool selected)
{
    constexpr int kPoints = 64;
    juce::Path path;
    bool first = true;
    for (int n = 0; n <= kPoints; ++n)
    {
        const float t   = (float)n / (float)kPoints;
        const float amp = applyEnvelope(shape, t);
        const float x   = bounds.getX() + t * bounds.getWidth();
        const float y   = bounds.getBottom() - amp * bounds.getHeight();
        if (first) { path.startNewSubPath(x, y); first = false; }
        else        path.lineTo(x, y);
    }
    g.setColour(selected ? juce::Colour(0xff7CF5C4) : juce::Colour(0xff5a5f72));
    g.strokePath(path, juce::PathStrokeType(1.2f));
}

void CurveEditor::mouseDown(const juce::MouseEvent& e)
{
    if (apvts_ == nullptr) return;
    for (int i = 0; i < kNumShapes; ++i)
    {
        if (thumbnailBounds_[i].contains(e.getPosition()))
        {
            if (auto* p = apvts_->getParameter(ParamIDs::envelopeShape))
                p->setValueNotifyingHost(p->convertTo0to1((float)i));
            break;
        }
    }
}
```
