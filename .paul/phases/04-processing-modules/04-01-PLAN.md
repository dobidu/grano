<objective>
## Goal
Implement the MOTION module (Wow, Flutter, Drift, Crackle) with bypass, wired to APVTS and the granular engine.

## Purpose
MOTION is the first of three optional processing modules (F4a/F4b/F4c). It adds analog-style instability — Wow and Flutter modulate per-grain pitch via slow LFOs, Drift adds a random-walk master pitch offset, and Crackle injects filtered pink noise into the output. All effects are zero-CPU when the module is bypassed.

## Output
- `Source/Modules/Motion.{h,cpp}` — standalone module class (no JUCE component, no UI)
- 8 new APVTS parameters in `Source/Parameters.{h,cpp}`
- `GranularEngine` reads a pitch-mod offset atomic from Motion
- `PluginProcessor` owns and drives Motion in `processBlock`
- `Tests/test_motion.cpp` — Catch2 unit tests
</objective>

```cpp
#pragma once
#include <juce_audio_basics/juce_audio_basics.h>
#include <atomic>
#include <cmath>

class Motion
{
public:
    void prepare(double sampleRate) noexcept;
    void reset()                    noexcept;

    void setParamPointers(std::atomic<float>* enabled,
                          std::atomic<float>* wowDepth,  std::atomic<float>* wowRate,
                          std::atomic<float>* flutDepth, std::atomic<float>* flutRate,
                          std::atomic<float>* drift,
                          std::atomic<float>* crackLevel, std::atomic<float>* crackColor) noexcept;

    // Called from audio thread in processBlock.
    // Updates pitchModSt_ and adds crackle to buffer.
    void processBlock(juce::AudioBuffer<float>& buffer) noexcept;

    // Read by GranularEngine::scheduleGrain() (scheduler thread).
    // Returns total pitch offset in semitones (Wow + Flutter + Drift).
    float getPitchModSemitones() const noexcept
    {
        return pitchModSt_.load(std::memory_order_relaxed);
    }

private:
    float nextPink()  noexcept;   // Paul Kellet pink noise, normalized ≈ [-1,1]
    void  updateLFOs(int numSamples) noexcept;

    double sampleRate_ { 44100.0 };

    // LFO phases (audio thread only, no sync needed)
    float wowPhase_     { 0.0f };
    float flutterPhase_ { 0.0f };

    // Drift random walk (audio thread only)
    float driftCurrent_ { 0.0f };
    float driftTarget_  { 0.0f };
    int   driftCounter_ { 0 };

    // Pink noise (Paul Kellet 7-stage, audio thread only)
    float b0_ {}, b1_ {}, b2_ {}, b3_ {}, b4_ {}, b5_ {}, b6_ {};

    // Color filter state (1-pole IIR, audio thread only)
    float filterState_ { 0.0f };

    // Cross-thread: audio thread writes, scheduler thread reads
    std::atomic<float> pitchModSt_ { 0.0f };

    // Param pointers (set once before processing)
    std::atomic<float>* pEnabled_     { nullptr };
    std::atomic<float>* pWowDepth_    { nullptr };
    std::atomic<float>* pWowRate_     { nullptr };
    std::atomic<float>* pFlutDepth_   { nullptr };
    std::atomic<float>* pFlutRate_    { nullptr };
    std::atomic<float>* pDrift_       { nullptr };
    std::atomic<float>* pCrackLevel_  { nullptr };
    std::atomic<float>* pCrackColor_  { nullptr };

    juce::Random rng_;

    JUCE_DECLARE_NON_COPYABLE_WITH_LEAK_DETECTOR(Motion)
};
```

**1d. Source/Modules/Motion.cpp**:

```cpp
#include "Motion.h"

static constexpr float kWowMaxCents     = 50.0f;   // ±50 cents at depth 1
static constexpr float kFlutterMaxCents = 10.0f;   // ±10 cents at depth 1
static constexpr float kDriftMaxCents   = 20.0f;   // ±20 cents at amount 1
static constexpr float kTwoPi          = 6.28318530718f;
static constexpr int   kDriftBlocksMin  = 200;     // min blocks between drift target changes
static constexpr int   kDriftBlocksMax  = 800;

void Motion::prepare(double sampleRate) noexcept
{
    sampleRate_ = sampleRate;
    reset();
}

void Motion::reset() noexcept
{
    wowPhase_ = flutterPhase_ = 0.0f;
    driftCurrent_ = driftTarget_ = 0.0f;
    driftCounter_ = kDriftBlocksMin;
    b0_ = b1_ = b2_ = b3_ = b4_ = b5_ = b6_ = 0.0f;
    filterState_ = 0.0f;
    pitchModSt_.store(0.0f, std::memory_order_relaxed);
}

void Motion::setParamPointers(std::atomic<float>* enabled,
                               std::atomic<float>* wowDepth,  std::atomic<float>* wowRate,
                               std::atomic<float>* flutDepth, std::atomic<float>* flutRate,
                               std::atomic<float>* drift,
                               std::atomic<float>* crackLevel, std::atomic<float>* crackColor) noexcept
{
    pEnabled_    = enabled;
    pWowDepth_   = wowDepth;  pWowRate_   = wowRate;
    pFlutDepth_  = flutDepth; pFlutRate_  = flutRate;
    pDrift_      = drift;
    pCrackLevel_ = crackLevel; pCrackColor_ = crackColor;
}

float Motion::nextPink() noexcept
{
    const float white = rng_.nextFloat() * 2.0f - 1.0f;
    b0_ = 0.99886f * b0_ + white * 0.0555179f;
    b1_ = 0.99332f * b1_ + white * 0.0750759f;
    b2_ = 0.96900f * b2_ + white * 0.1538520f;
    b3_ = 0.86650f * b3_ + white * 0.3104856f;
    b4_ = 0.55000f * b4_ + white * 0.5329522f;
    b5_ = -0.7616f * b5_ - white * 0.0168980f;
    const float pink = (b0_ + b1_ + b2_ + b3_ + b4_ + b5_ + b6_ + white * 0.5362f) * 0.11f;
    b6_ = white * 0.115926f;
    return juce::jlimit(-1.0f, 1.0f, pink);
}

void Motion::processBlock(juce::AudioBuffer<float>& buffer) noexcept
{
    if (pEnabled_ == nullptr)
        return;
    const bool enabled = pEnabled_->load(std::memory_order_relaxed) > 0.5f;
    if (!enabled)
    {
        pitchModSt_.store(0.0f, std::memory_order_relaxed);
        return;
    }

    const int    N           = buffer.getNumSamples();
    const float  sr          = (float)sampleRate_;
    const float  wowDepth    = pWowDepth_   ? pWowDepth_->load(std::memory_order_relaxed)   : 0.0f;
    const float  wowRate     = pWowRate_    ? pWowRate_->load(std::memory_order_relaxed)    : 0.5f;
    const float  flutDepth   = pFlutDepth_  ? pFlutDepth_->load(std::memory_order_relaxed)  : 0.0f;
    const float  flutRate    = pFlutRate_   ? pFlutRate_->load(std::memory_order_relaxed)   : 8.0f;
    const float  driftAmt    = pDrift_      ? pDrift_->load(std::memory_order_relaxed)      : 0.0f;
    const float  crackLevelDb= pCrackLevel_ ? pCrackLevel_->load(std::memory_order_relaxed) : -60.0f;
    const float  crackColor  = pCrackColor_ ? pCrackColor_->load(std::memory_order_relaxed) : 0.5f;

    // ── Wow (end-of-block sample for pitch offset) ────────────────────────────
    const float wowIncr = kTwoPi * wowRate / sr;
    wowPhase_ = std::fmod(wowPhase_ + wowIncr * (float)N, kTwoPi);
    const float wowCents = std::sin(wowPhase_) * wowDepth * kWowMaxCents;

    // ── Flutter (sine + 10% white noise) ─────────────────────────────────────
    const float flutIncr = kTwoPi * flutRate / sr;
    flutterPhase_ = std::fmod(flutterPhase_ + flutIncr * (float)N, kTwoPi);
    const float flutCents = (0.9f * std::sin(flutterPhase_)
                             + 0.1f * (rng_.nextFloat() * 2.0f - 1.0f))
                            * flutDepth * kFlutterMaxCents;

    // ── Drift random walk ─────────────────────────────────────────────────────
    if (--driftCounter_ <= 0)
    {
        driftCounter_ = kDriftBlocksMin
            + rng_.nextInt(kDriftBlocksMax - kDriftBlocksMin);
        driftTarget_ = (rng_.nextFloat() * 2.0f - 1.0f) * driftAmt * kDriftMaxCents;
    }
    const float driftSmooth = 0.0005f;  // per-block smoothing coefficient
    driftCurrent_ += (driftTarget_ - driftCurrent_) * driftSmooth;
    const float driftCents = driftCurrent_;

    // ── Publish pitch offset to scheduler thread ──────────────────────────────
    const float totalCents = wowCents + flutCents + driftCents;
    pitchModSt_.store(totalCents / 100.0f, std::memory_order_relaxed);  // → semitones

    // ── Crackle ───────────────────────────────────────────────────────────────
    if (crackLevelDb <= -59.9f)
        return;

    const float gain = juce::Decibels::decibelsToGain(crackLevelDb, -60.0f);
    // Color: 0 = LPF (warm), 1 = HPF (bright). Map to 1-pole coefficient.
    // coeff near 1 = LPF pass, coeff near 0 = HPF pass.
    const float lpfCoeff = 1.0f - crackColor * 0.97f;

    for (int ch = 0; ch < buffer.getNumChannels(); ++ch)
    {
        float* data = buffer.getWritePointer(ch);
        for (int n = 0; n < N; ++n)
        {
            float pink = nextPink();
            // 1-pole IIR: LPF when coeff high, acts as HPF when we subtract filtered
            filterState_ = lpfCoeff * filterState_ + (1.0f - lpfCoeff) * pink;
            // crackColor=0 → full LPF output (warm); crackColor=1 → (pink - filtered) = HPF
            const float colored = juce::jmap(crackColor, filterState_, pink - filterState_);
            data[n] += colored * gain;
        }
    }
}
```
