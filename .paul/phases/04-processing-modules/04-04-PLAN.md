## Goal
Implement the PATTERN module: Trigger Mode (Free / Sync / Euclidean / Audio-driven), Probability, Reverse Probability, Quantize (pitch to scale), and Spray (randomness macro). Wire to APVTS and GranularEngine scheduler.

## Purpose
PATTERN is the third F4 module. Unlike Motion (pitch modulation) and Color (post-engine DSP), PATTERN affects grain scheduling: when grains fire, whether they play reversed, and their pitch alignment. The module is zero-effect when bypassed.

**quantizeToScale()**:
```cpp
float Pattern::quantizeToScale(float pitchSt, int scaleIdx) noexcept
{
    if (scaleIdx < 0 || scaleIdx >= kNumScales)
        return pitchSt;
    const Scale& sc = kScales[scaleIdx];
    if (sc.count <= 0) return pitchSt;

    // Fold to [0, 12) modulo-12 offset within the octave
    const float octave = std::floor(pitchSt / 12.0f);
    float mod12 = pitchSt - octave * 12.0f;  // [0, 12)

    // Find nearest scale note (handle wrap-around: also consider note[0]+12)
    float bestDist = 999.0f;
    float bestNote = 0.0f;
    for (int i = 0; i < sc.count; ++i)
    {
        const float note = (float)sc.notes[i];
        float dist = std::abs(mod12 - note);
        if (dist < bestDist) { bestDist = dist; bestNote = note; }
    }
    // Check wrap: first note + 12 (e.g., C above B)
    {
        const float note = (float)sc.notes[0] + 12.0f;
        float dist = std::abs(mod12 - note);
        if (dist < bestDist) { bestDist = dist; bestNote = note; }
    }
    return octave * 12.0f + bestNote;
}
```

```cpp
{
    if (pEnabled_ == nullptr || pEnabled_->load(std::memory_order_relaxed) < 0.5f)

    float pitch = basePitchSt;

    // Apply quantize
    const int scaleIdx = pQuantizeScale_

    const float sprayAmt = pSpray_

        pitch += (rng_.nextFloat() * 2.0f - 1.0f) * sprayAmt * kMaxSprayPitchSt;

    return pitch;
}
```

**Euclidean Bjorklund implementation**:
```cpp
Pattern::EuclidPattern Pattern::buildEuclidean(int pulses, int steps, int rotation) noexcept
{
    EuclidPattern pat{};
    if (steps <= 0 || pulses <= 0) return pat;
    pulses = std::min(pulses, steps);

    // Standard Bjorklund distribution: spread pulses as evenly as possible across steps.
    // Compute using the Euclidean remainder sequence (Toussaint 2005).
    std::array<int, 32> counts{};
    std::array<int, 32> remainders{};

    int numGroups  = pulses;
    int groupSize  = steps / pulses;
    int rem        = steps % pulses;

    for (int i = 0; i < pulses; ++i)      counts[i] = groupSize + (i < rem ? 1 : 0);
    for (int i = 0; i < pulses; ++i)      remainders[i] = (i < rem) ? 1 : 0;

    // Repeatedly fold remainders into the main groups
    while (rem > 1)
    {
        const int newGroups = std::min(numGroups, rem);
        const int oldRem    = rem;
        rem = numGroups - rem;
        numGroups = newGroups;
        for (int i = 0; i < newGroups; ++i)
        {
            counts[i] += remainders[i];
            remainders[i] = (i < oldRem) ? 1 : 0;
        }
    }

    // Expand: each group produces 1 hit followed by (counts[i]-1) rests
    int pos = 0;
    for (int g = 0; g < numGroups && pos < steps; ++g)
        for (int k = 0; k < counts[g] && pos < steps; ++k, ++pos)
            pat[pos] = (k == 0);

    // Apply rotation (left-rotate the pattern by `rotation` steps)
    if (rotation > 0)
    {
        rotation = rotation % steps;
        EuclidPattern rotated{};
        for (int i = 0; i < steps; ++i)
            rotated[i] = pat[(i + rotation) % steps];
        pat = rotated;
    }

    return pat;
}

void Pattern::rebuildEuclidIfNeeded() noexcept
{
    const int p = pEuclidPulses_  ? (int)pEuclidPulses_->load(std::memory_order_relaxed) : 4;
    const int s = pEuclidSteps_   ? (int)pEuclidSteps_->load(std::memory_order_relaxed)  : 8;
    const int r = pEuclidRotation_? (int)pEuclidRotation_->load(std::memory_order_relaxed): 0;
    if (p == cachedPulses_ && s == cachedSteps_ && r == cachedRotation_)
        return;
    euclidPattern_   = buildEuclidean(p, s, r);
    euclidPatternLen_ = std::max(s, 1);
    euclidStep_      = 0;
    cachedPulses_    = p;
    cachedSteps_     = s;
    cachedRotation_  = r;
}
```

**processBlock() — Sync BPM + Audio transient detection**:
```cpp
void Pattern::processBlock(const juce::AudioBuffer<float>& inputBuffer, double bpm) noexcept
{
    if (pEnabled_ == nullptr || pEnabled_->load(std::memory_order_relaxed) < 0.5f)
        return;

    const int mode = pTriggerMode_
        ? (int)pTriggerMode_->load(std::memory_order_relaxed) : 0;

    // ── Sync interval ────────────────────────────────────────────────────────
    if (mode == 1 /* Sync */ && bpm > 0.0)
    {
        const int divIdx = pSyncDivision_
            ? (int)pSyncDivision_->load(std::memory_order_relaxed) : 1;
        // Division table: 1/4, 1/8, 1/16, 1/32, 1/64, 1/4T, 1/8T, 1/16T, 1/4D, 1/8D
        // Multiplier relative to quarter note (1 beat = 60/bpm seconds)
        static constexpr float kDivMult[] = {
            1.0f, 0.5f, 0.25f, 0.125f, 0.0625f,  // straight
            2.0f/3.0f, 1.0f/3.0f, 1.0f/6.0f,      // triplet
            1.5f, 0.75f                             // dotted
        };
        const int clampedIdx = juce::jlimit(0, 9, divIdx);
        const float beatSec  = (float)(60.0 / bpm);
        syncIntervalAtomic_.store(beatSec * kDivMult[clampedIdx],
                                  std::memory_order_relaxed);
    }

    // ── Audio-driven transient detection ────────────────────────────────────
    if (mode == 3 /* Audio */ && inputBuffer.getNumChannels() > 0)
    {
        const float sens = pTransientSens_
            ? pTransientSens_->load(std::memory_order_relaxed) : 0.5f;

        const float* ch = inputBuffer.getReadPointer(0);
        const int    N  = inputBuffer.getNumSamples();

        // Attack/release energy follower (per-block approximation)
        float blockEnergy = 0.0f;
        for (int n = 0; n < N; ++n) blockEnergy += ch[n] * ch[n];
        blockEnergy /= std::max(N, 1);

        const float kAttack  = 0.9f;
        const float kRelease = 0.05f;
        energyFollower_ = blockEnergy > energyFollower_
            ? kAttack * energyFollower_ + (1.0f - kAttack) * blockEnergy
            : kRelease * energyFollower_ + (1.0f - kRelease) * blockEnergy;

        // Threshold: normalized to sensitivity. Higher sens = more triggers.
        // Onset: current energy significantly above follower (ratio-based).
        const float threshold = 0.3f - sens * 0.25f;  // [0.05 .. 0.30]
        const float ratio = energyFollower_ > 1e-10f
            ? blockEnergy / energyFollower_ : 0.0f;

        if (transientCooldown_ > 0)
            transientCooldown_ -= N;

        if (ratio > (1.0f + threshold) && transientCooldown_ <= 0)
        {
            transientFired_.store(true, std::memory_order_relaxed);
            transientCooldown_ = (int)(kTransientCooldownMs * 0.001f * (float)sampleRate_);
        }

        lastEnergy_ = blockEnergy;
    }
}
```

```cpp
float Pattern::getNextIntervalSec() noexcept
{
    if (!isSchedulerOverrideActive()) return -1.0f;

    const int mode = pTriggerMode_
        ? (int)pTriggerMode_->load(std::memory_order_relaxed) : 0;

    switch (mode)
    {
        case 1: // Sync
            return syncIntervalAtomic_.load(std::memory_order_relaxed);

        case 2: // Euclidean
        {
            rebuildEuclidIfNeeded();
            const int p = pEuclidPulses_ ? (int)pEuclidPulses_->load(std::memory_order_relaxed) : 4;
            const int s = pEuclidSteps_  ? (int)pEuclidSteps_->load(std::memory_order_relaxed)  : 8;
            const float density = 10.0f; // fallback
            // Step duration: 1 full pattern cycle = steps * (1/density) seconds
            // But pattern fires at each step regardless of whether it's a hit;
            // shouldFireGrain() gates the actual grain spawn.
            const float patternCycleSec = (float)s / std::max(density, 0.1f);
            const float stepSec = patternCycleSec / (float)s;
            (void)p;
            return stepSec;
        }

        case 3: // Audio-driven
            // Poll quickly; shouldFireGrain() fires only when transient detected.
            return kPollingIntervalSec;

        default:
            return -1.0f;
    }
}
```

```cpp
        ? (int)pTriggerMode_->load(std::memory_order_relaxed) : 0;

    {
        rebuildEuclidIfNeeded();
        const bool hit = euclidPattern_[euclidStep_ % euclidPatternLen_];

        if (!fired) return false;

        ? pProbability_->load(std::memory_order_relaxed) : 1.0f;
    return rng_.nextFloat() < prob;
}
```
